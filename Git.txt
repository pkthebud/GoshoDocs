        git-add(1)

        git-am(1)
            Apply a series of patches from a mailbox.

        // rightgif https://media.giphy.com/media/Q6gPyUYrCk76g/giphy.gif
        git-archive(1)
            Create an archive of files from a named tree.

        // pic https://static.pexels.com/photos/2371/person-looking-searching-clean.jpg
        git-bisect(1) 
            Find by binary search the change that introduced a bug.

        git-branch(1)
            List, create, or delete branches.

        //pic https://static.pexels.com/photos/1091/coffee-cup-table-morning.jpg
        git-bundle(1)
            Move objects and refs by archive.

    C
        // pic https://static.pexels.com/photos/6490/people-hotel-bar-drinks.jpg
        git-checkout(1)
            Checkout a branch or paths to the working tree.

        // pic https://static.pexels.com/photos/8343/food-cherries.jpg
        git-cherry-pick(1)
            Apply the changes introduced by some existing commits.
            
        // pic https://static.pexels.com/photos/4316/technology-computer-chips-gigabyte.jpg
        git-citool(1)
            Graphical alternative to git-commit.

        // rightgif https://media.giphy.com/media/lT0wsL6b6tbfq/giphy.gif
        git-clean(1)
            Remove untracked files from the working tree.

        // gif https://media.giphy.com/media/K6QMZYCuavr4Q/giphy.gif
        git-clone(1)
            Clone a repository into a new directory.

        // pic https://static.pexels.com/photos/8859/drink-beer.jpg
        git-commit(1)
            Record changes to the repository.

        // pic https://static.pexels.com/photos/9137/night-vintage-music-bokeh.jpg
        git-describe(1)
            Show the most recent tag that is reachable from a commit.
        
        // pic https://static.pexels.com/photos/539/man-person-legs-grass.jpg
        git-diff(1)
            Show changes between commits, commit and working tree, etc.
    F
        // pic https://static.pexels.com/photos/1173/animal-dog-pet-cute.jpg
        git-fetch(1)
            Download objects and refs from another repository.

        // pic https://static.pexels.com/photos/4627/hands-creative-grass-moss.jpg
        git-format-patch(1)
            Prepare patches for e-mail submission.
    G
        // leftgif https://media.giphy.com/media/hS2Oh3m0CkhFe/giphy.gif
        git-gc(1)
            Cleanup unnecessary files and optimize the local repository.

        // rightgif https://media3.giphy.com/media/D3NlvPQaDhJ0A/200.gif
        git-grep(1)
            Print lines matching a pattern.

        // leftgif https://media2.giphy.com/media/dnYQ7p67dX6Za/200.gif

        git-init(1)
            Create an empty Git repository or reinitialize an existing one.
    L
        // pic https://static.pexels.com/photos/792/wood-landscape-mountains-nature.jpg

            Show commit logs.
    M
        // pic https://static.pexels.com/photos/4634/people-street-sidewalk-jeans.jpg
        git-merge(1)
            Join two or more development histories together.
    
        // pic https://static.pexels.com/photos/5358/sea-beach-holiday-vacation.jpg

            Add or inspect object notes.

        // pic https://static.pexels.com/photos/4905/door-rustic.jpg
        git-pull(1)

        // pic https://static.pexels.com/photos/6620/pexels-photo.jpeg

        // rightgif https://media2.giphy.com/media/aOzFWT8lRj41q/200.gif
